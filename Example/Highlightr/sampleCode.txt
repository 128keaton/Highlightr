import Foundation

@objc class Person: Entity {
    var name: String!
    var age:  Int!

    init(name: String, age: Int) {
        /* /* ... */ */
        self.name = name
        self.age = age
    }

    // Return a descriptive string for this person
    func description(offset: Int = 0) -> String {
        return "\(name) is \(age + offset) years old"
    }

    func asd() -> String {
        description()
    }
}
